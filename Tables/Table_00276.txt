    PROCEDURE GetCurrencyCode@2() : Code[10];
    VAR
      BankAcc@1000 : Record 270;
    BEGIN

        EXIT(BankAcc."Currency Code");

      IF BankAcc.GET("Bank Account No.") THEN
        EXIT(BankAcc."Currency Code");

      EXIT('');
    END;
