      DataStream@1001 : OutStream;

      CLEAR(Error);

    END;

    PROCEDURE GetError@2() ErrorMsg : Text;
    VAR
      DataStream@1000 : InStream;
    BEGIN
      ErrorMsg := '';
      CALCFIELDS(Error);
      IF Error.HASVALUE THEN BEGIN
        Error.CREATEINSTREAM(DataStream);
        DataStream.READ(ErrorMsg);
      END;
    END;

    END;
